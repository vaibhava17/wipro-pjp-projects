1. SELECT EMPNO,
   E.ENAME,
   E.SAL,
   D.DNAME,
   D.LOC
   FROM EMP E
   NATURAL JOIN DEPT D;

2. SELECT E.JOB,E.MGR,E.SAL,E.COMM,D.DNAME 
   FROM EMP E ,DEPT D
   WHERE E. DEPTID = D. DEPTID AND E.JOB = 'SALESMAN';

3. SELECT E.ENAME,E.JOB,D.DEPTNO,D.DNAME
   FROM EMP E ,DEPT D
   WHERE E. DEPTID = D. DEPTID AND D.LOC = 'DALLAS';

4. SELECT ENAME "EMPLOYEE",EMPID "EMP#" ,M_NAME "MANAGER",MGRID "MGR#" FROM EMP.E1,EMP.E2 
   WHERE E1.EMPID = E2.MGRID;

5. SELECT ENAME "EMPLOYEE",EMPID "EMP#" ,M_NAME "MANAGER",MGRID "MGR#" FROM EMP.E1 LEFT OUTER JOIN EMP.E2 
   ON E1.EMPID = E2.MGRID AND ORDER BY EMPID;

6. SELECT DISTINCT E.ENAME, E.JOB, D.DNAME, E.SAL, S.GRADE
   FROM EMP E, DEPT D, SALGRADE S
   WHERE E.DEPTNO = D.DEPTNO ORDER BY S.GRADE;

7. SELECT E.ENAME,D.DNAME
   FROM EMP E
   LEFT JOIN DEPT D ON E.DEPTID=D.DEPTID;

8. SELECT E.ENAME,E.HIREDATE,M.ENAME,M.HIREDATE
   from EMP E,EMP M 
   WHERE E.EMPID = M. MGRID AND M.HIREDATE> E.HIREDATE;

9. SELECT EMPNO,ENAME,DNAME,LOC
   FROM EMP
   JOIN DEPT USING (DEPTNO)
   WHERE EMP.JOB = 'CLERK'

10. SELECT EMPID,ENAME,DNAME,LOC
    FROM EMP
    JOIN DEPT USING (DEPTNO)
    WHERE EMP.SAL > 2000;

11. SELECT E.EMPNO,E.ENAME,D.DNAME,D.LOC
    FROM EMP E
    LEFT JOIN DEPT D 
    ON E.DEPTID = D.DEPTID ;

12. SELECT E.ENAME,D.DNAME
    FROM EMP E
    RIGHT JOIN DEPT D ON E.DEPTID=D.DEPTID;

13. SELECT E.ENAME,D.DNAME
    FROM EMP E
    FULL JOIN DEPT D ON E.DEPTNO=D.DEPTNO;
