CREATE ADD_DEPT(DNO, NAME, LO)DEPT
RETURN DEPT;
BEGIN
SELECT COUNT (*) FORM DEPT
INSERT 
DNO=DEPT.DNO;
DNAME= DEPT.DNAME;
LO= DEPT.LO;
GROUP BY[DNO],[NAME],[LO]
HAVING COUNT(*)>1;
DBMS_OUTPUT.PUT('DUPLICATE ROW IS THERE :');
END;
CREATE UPDATE_DEPT(DNO, NAME, LO)DEPT
BEGIN
SELECT COUNT (*) FROM DEPT
UPDATE
DNO=DEPT.DNO;
DNAME= DEPT.DNAME;
LO=DEPT.LO;
FOR DEPTNO='';
IF
DEPTNO='' ;
THEN 
DBMS_OUTPUT.PUT('INFORMATION IS UPDATED');
ELSE
DBMS_OUTPUT.PUT('THAT DEPTNO IS NOT EXIST');
END;
CREATE DELETE_DEPT(DNO, NAME, LO)DEPT
BEGIN
SELECT COUNT (*) FORM DEPT
DELETE
DNO=DEPT.DNO;
DNAME= DEPT.DNAME;
LO=DEPT.LO;
FOR DEPTNO='';
IF DEPTNO='';
THEN DBMS_OUTPUT.PUT('ENTIRE ROW IS DELETED');
ELSE
DBMS_OUTPUT.PUT('THAT DEPTNO IS NOT EXIST');
END;
/
